 //Variable reference for the class
        private static UniversalAudio _universalAudio;
        private static bool _isPlayingTrack1;

        public void SwapAudio(string newClipName)
        {
            StopAllCoroutines();
            StartCoroutine(FadeTrack(newClipName, true));
            if (_isPlayingTrack1)
            {
                _audioSource2.clip = Resources.Load<AudioClip>(newClipName);
                _audioSource2.time = _audioClipTimerDictionary[newClipName];
                _audioSource2.Play();
                _audioSource1.Stop();
            }
            else
            {
                _audioSource1.clip = Resources.Load<AudioClip>(newClipName);
                _audioSource1.time = _audioClipTimerDictionary[newClipName];
                _audioSource1.Play();
                _audioSource2.Stop();
            }

            _isPlayingTrack1 = !_isPlayingTrack1;
        }

 public void ReturnToDefault()
        {
            SwapAudio("Swimming");
        }
